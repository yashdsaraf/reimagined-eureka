import {Component} from '@angular/core'

import {DocItem, DOCS} from './docs-content'

@Component({
  selector: 'app-docs',
  templateUrl: './docs.component.html',
  styleUrls: ['./docs.component.sass']
})
export class DocsComponent {

  menu: DocItem[]
  content: string

  constructor() {
    this.menu = DOCS
    this.content = "<h1>Hello</h1>"
  }

  loadUrlIntoDiv(url: string) {
    console.log(url)
  }

}

<div id="container">
  <div class="ui vertical menu" id="menu">
    <div class="header item">
      <h1 class="ui left aligned header" >Docs</h1>
    </div>
    <div class="ui divider"></div>
    <div *ngFor="let item of menu" class="item">
      <sui-accordion>
        <sui-accordion-panel>
          <div class="menu" title>
            {{item.title}}
            <i class="ui icon dropdown"></i>
          </div>
          <div class="info-content" content>
            <div *ngFor="let child of item.children" class="item" (click)="loadUrlIntoDiv(child.url)">{{child.title}}</div>
          </div>
        </sui-accordion-panel>
      </sui-accordion>
    </div>
  </div>
  <div class="ui basic segment" id="target-div" [innerHTML]="content | sanitizeHtml">
  </div>
</div>
